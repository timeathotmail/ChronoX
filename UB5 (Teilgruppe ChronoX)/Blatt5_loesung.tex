public abstract class Command{
	
	protected char changedChar; // Wird in den Klassen Edit UND Del ben√∂tigt
	
	public abstract void execute();
	public abstract void undo();
	public abstract void redo();

}

public class Edit extends Command{
	
	public Edit(char c){
		changedChar = c;
	}
	
	public void execute(){
		TextBuffer.add(changedChar);
	}
	
	public void undo(){
		TextBuffer.del();
	}
	
	public void redo(){
		execute();
	}

}

public class Del extends Command{
	
	public void execute(){
		changedChar = TextBuffer.del();
	}
	
	public void undo(){
		TextBuffer.add(changedChar);
	}
	
	public void redo(){
		execute();
	}

}
\end{lstlisting}
